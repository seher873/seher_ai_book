## Step 6: Create Glossary Search Component

Create a React component for glossary search functionality:

```bash
mkdir -p src/components/Glossary

cat > src/components/Glossary/GlossarySearch.jsx << 'EOF'
import React, { useState, useEffect } from 'react';

const GlossarySearch = ({ terms }) => {
  const [query, setQuery] = useState('');
  const [filteredTerms, setFilteredTerms] = useState([]);

  useEffect(() => {
    if (query.trim() === '') {
      setFilteredTerms([]);
    } else {
      const results = terms.filter(term =>
        term.term.toLowerCase().includes(query.toLowerCase()) ||
        term.definition.toLowerCase().includes(query.toLowerCase())
      );
      setFilteredTerms(results.slice(0, 10)); // Limit to 10 results
    }
  }, [query, terms]);

  return (
    <div className="glossary-search">
      <input
        type="text"
        placeholder="Search glossary terms..."
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        className="search-input"
      />
      {query && filteredTerms.length > 0 && (
        <ul className="search-results">
          {filteredTerms.map((term, index) => (
            <li key={index}>
              <a href={`#${term.id}`}>{term.term}</a>
              <p>{term.definition.substring(0, 100)}...</p>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
};

export default GlossarySearch;
EOF
```
